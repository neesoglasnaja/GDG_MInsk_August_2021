Contact Picker API

const supported = ('contacts' in navigator
   && 'ContactsManager' in window);
const props = ['name', 'email', 'tel', 'address', 'icon'];
const opts = { multiple: true };
 
navigator.contacts.select(props, opts)
  .then(contacts => { ... })
  .catch(e => { ... })
